# Exibir configurações do git

    git config --list --show-origin
    git config --global --list
    git config --list

# Setar informações pessoais

    git config --global user.name "nome qualquer"
    git config --global user.email email@teste.com.br

# Iniciar repositorio local

    git init

# Clonando repositório

    git clone URL nomeRepositorio

# Verificar estado dos arquivos e estado resumido

    git status
    git status -s
    git status --short

# Enviar arquivo para stage area

    git add [nomeArquivo | nomepasta | * para tudo]

# Exibe diferenças dos arquivos

    git diff            Compara a chage area com stage area
    git diff --staged   Compara a stage area com ultimo commit, pressionar a letra Q para sair
