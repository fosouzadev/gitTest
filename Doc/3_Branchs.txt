





# Publicar branch local em servidor remoto

    git push [origin | nomeLocalRemoto] nomeBranch                 Não cria rastreamento com branch remota
    git push --set-upstream [origin | localRemoto] nomeBranch      Cria um rastreamento com branch remota

# Publicar alterações de branch local em branch remota

    git push   (no repositorio selecionado)
    git push [origin | nomeLocalRemoto] [nomeBranch]
