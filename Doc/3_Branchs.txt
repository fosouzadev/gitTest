# Criar um branch

    git branch nomeBranch

# Alterar de branch
    
    git checkout nomeBranch

# Criar e alterar para nova branch

    git checkout -b nomeBranch

# Merge entre branchs: use o checkout para selecionar a branch destino

    git merge nomeBranchOrigem

# Publicar branch local em servidor remotoj

    git push --set-upstream [origin | localRemoto] nomeBranch
