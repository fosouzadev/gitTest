# Exibir local das branchs remotas

    git remote
    git remote -v

# Exibir branchs remotas

    git ls-remote
    git remote show [origin | nomeLocal]

# Baixa as alterações do repositório remoto, mas não faz o merge

    git fetch [origin | nomeLocalRemoto]

# Baixa e faz merge das alterações do repositório remoto

    git pull 

# Enviar commits para o repositório remoto

    git push   (no repositorio selecionado)
    git push [origin | nomeLocalRemoto] [nomeBranch]

# Listar branchs locais

    git branch
    git branch -v                       Exibe ultimo commit de cada branch
    git branch --merged                 Exibe branchs que ja foram mescladas com a branch selecionada
    git branch --no-merged              Exibe branchs que ainda não foram mescladas com a branch selecionada
    git branch --no-merged nomeBranch   Exibe o que não foi mesclado com nomeBranch

# Criar uma branch local

    git branch nomeBranch

# Alterar de branch
    
    git checkout nomeBranch

# Criar e alterar para a nova branch

    git checkout -b nomeNovaBranch                          Cria a partir de branch local
    git checkout -b nomeNovaBranch nomeRemoto/nomeBranch    Cria a partir de branch remota

# Excluir branch local

    git branch -d nomeBranch
    git branch -D nomeBranch        Força a exclusão de branch que não foi feito merge ainda

# Merge entre branchs : use o checkout para selecionar a branch destino

    git merge nomeBranchOrigem          Merge local
    git merge nomeRemoto/nomeBranch     Merge com brachs remotas

# Publicar branch local em servidor remoto

    git push --set-upstream [origin | localRemoto] nomeBranch

# Publicar alterações de branch local em servidor remoto

    git push   (no repositorio selecionado)
    git push [origin | nomeLocalRemoto] [nomeBranch]
