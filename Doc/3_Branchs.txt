# Listar branchs locais

    git branch

# Criar uma branch local

    git branch nomeBranch

# Alterar de branch
    
    git checkout nomeBranch

# Criar e alterar para a nova branch local

    git checkout -b nomeBranch

# Excluir branch local

    git branch -d nomeBranch

# Merge entre branchs locais: use o checkout para selecionar a branch destino

    git merge nomeBranchOrigem

# Publicar branch local em servidor remoto

    git push --set-upstream [origin | localRemoto] nomeBranch

# Publicar alterações de branch local em servidor remoto

    git push
