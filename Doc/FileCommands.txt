# Verificar estado dos arquivos e estado resumido

    git status
    git status [-s | --short]

# Enviar arquivo para stage area

    git add [nomeArquivo | nomepasta | *]

# Retirar arquivos da stage area

    git restore --staged [nomeArquivo | nomePasta | *]
    git reset HEAD nomeArquivo
    git reset HEAD *

# Exibe diferenças dos arquivos

    git diff                Compara a chage area com ultimo commit, caso o arquivo esteja na change e stage, tambem exibe alterações
    git diff --staged       Compara a stage area com ultimo commit, pressionar a letra Q para sair
    git diff --name-only    Compara a chage area com ultimo commit, caso o arquivo esteja na change e stage, tambem exibe o nome

# Excluir arquivo

    rm [nomeArquivo | nomePasta]

# Mover ou renomear arquivos

    git mv nomeArquivo.ext nomeNovoArquivo.ext      Renomeia
    git mv pasta/arquivo.ext pasta/arquivo.ext      Move
    git mv pasta/arquivo.ext pasta/novoArquivo.ext  Move e renomeia

# Reverter alterações em arquivos da change area

    git checkout [nomeArquivo | nomePasta | *]
